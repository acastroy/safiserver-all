#!/bin/sh
#
# safiserverd    Starts, Stops and Reloads SafiServer.
#
# chkconfig: 2345 90 60
# description: SafiServer process start
# processname: safiserver
# pidfile:  /SafiServer.pid

# Directory where SafiServer is installed
SAFISERVER_DIR=

# Source function library.
. /etc/rc.d/init.d/functions


if ! [ -x $SAFISERVER_DIR/installserver.sh ] ; then
	echo "ERROR: $SAFISERVER_DIR/installserver.sh not found"
	exit 0
fi

RETVAL=0

start() {
	# Start daemons.
	echo -n $"  Running safiserver wrapper service..."
	echo ""
	cd $SAFISERVER_DIR
	./safiserver start
	touch /var/lock/safiserver
    return
}

stop() {
	# Stop daemons.
	echo -n "  Shutting down safiserver wrapper service..."
	cd $SAFISERVER_DIR
	./safiserver stop
    rm -f /var/lock/safiserver
    return 
}

restart() {
	stop
	start
}

reload() {
	stop
	start
}

# See how we were called.
case "$1" in
  start)
	start
	;;
  stop)
	stop
	;;
  restart)
	restart
	;;
  reload)
	reload
	;;
  condrestart)
	[ -f /var/lock/safiserver ] && restart || :
	;;
  *)
	echo "Usage: safiserverd {start|stop|restart|reload|condrestart}"
	exit 1
esac

exit $?
